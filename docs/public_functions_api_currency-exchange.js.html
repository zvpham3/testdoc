/**
 * Module for fetching and displaying live currency exchange rates using the ExchangeRate-API.
 *
 * Provides functionality for selecting different target currencies and dynamically
 * updating the displayed conversion rate from the base currency.
 *
 * @module API/CurrencyExchange
 */

const EXCHANGE_API_KEY = "df2e3ff089d04850763f2aa7"; 
const BASE_CURRENCY = "USD";
const TARGET_SELECT_ID = "currency-selector";
const RATE_DISPLAY_ID = "exchange-rate";
const EXCHANGE_API_URL = `https://v6.exchangerate-api.com/v6/${EXCHANGE_API_KEY}/latest/${BASE_CURRENCY}`;

/**
 * Fetches the latest exchange rate for the given target currency
 * and updates the HTML display element with the result.
 * 
 * @async
 * @function fetchExchangeRate
 * @memberof module:API/CurrencyExchange
 * @param {string} targetCurrency - The currency code to convert the base currency into (e.g., 'EUR').
 * @returns {Promise<void>} Updates the DOM element with the conversion rate.
 */
async function fetchExchangeRate(targetCurrency) {
    const API_URL = `https://v6.exchangerate-api.com/v6/${EXCHANGE_API_KEY}/latest/${BASE_CURRENCY}`;

    try {
        const response = await fetch(API_URL);
        if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);

        const data = await response.json();
        if (data.result === "success") {
            const rate = data.conversion_rates[targetCurrency];
            document.getElementById(RATE_DISPLAY_ID).textContent =
                `1 ${BASE_CURRENCY} = ${rate} ${targetCurrency}`;
        } else {
            document.getElementById(RATE_DISPLAY_ID).textContent = "Error fetching rate";
            console.error("Exchange API error:", data["error-type"]);
        }
    } catch (err) {
        console.error("Error fetching exchange rate:", err);
        document.getElementById(RATE_DISPLAY_ID).textContent = "Error fetching rate";
    }
}

/**
 * Initializes the exchange rate module when the DOM loads.
 *
 * @function initCurrencyExchange
 * @memberof module:API/CurrencyExchange
 */
document.addEventListener("DOMContentLoaded", () => {
    const selector = document.getElementById(TARGET_SELECT_ID);

    if (!selector) return;

    fetchExchangeRate(selector.value);

    selector.addEventListener("change", () => {
        fetchExchangeRate(selector.value);
    });
});
