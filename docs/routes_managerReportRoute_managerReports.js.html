/**
 * Express router handling Manager Reports routes.
 * 
 * This module serves as the main entry point for all manager reporting
 * functionality, including sales reports, X/Z reports, restock reports,
 * and product usage reports.
 *
 * @module routes/managerReports
 */

const express = require('express');
const router = express.Router();
const db = require('../../db');

const salesReportRouter = require('./salesReport');
const XandZReportRouter = require('./XandZReport');
const restockReportRouter = require('./restockReport');
const productusageReportRouter = require('./productUsageReport');

/**
 * GET /manager/managerReports  
 * Renders the Manager Reports main menu page.
 *
 * @name GET /
 * @function
 * @param {express.Request} req - Express request object
 * @param {express.Response} res - Express response object
 */
router.get('/', async (req, res) => {
    res.render('manager_reports/managerReports');
});

/**
 * Mounts the Sales Report router.
 * @name /salesReport
 */
router.use('/salesReport', salesReportRouter);

/**
 * Mounts the X and Z Reports router.
 * @name /xAndzReport
 */
router.use('/xAndzReport', XandZReportRouter);

/**
 * Mounts the Restock Report router.
 * @name /restock
 */
router.use('/restock', restockReportRouter);

/**
 * Mounts the Product Usage Report router.
 * @name /productUsageReport
 */
router.use('/productUsageReport', productusageReportRouter);


module.exports = router;
